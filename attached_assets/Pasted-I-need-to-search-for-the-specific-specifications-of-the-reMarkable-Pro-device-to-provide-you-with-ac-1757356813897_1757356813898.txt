// Responsive typography for narrow screen
function adjustTypography() {
  const viewportWidth = 91; // mm
  const baseFontSize = Math.max(6, Math.min(8, viewportWidth / 15));
  
  document.documentElement.style.setProperty('--base-font', `${baseFontSize}pt`);
  document.documentElement.style.setProperty('--small-font', `${baseFontSize * 0.8}pt`);
  document.documentElement.style.setProperty('--tiny-font', `${baseFontSize * 0.7}pt`);
  
  // Ensure minimum readability
  const allText = document.querySelectorAll('*');
  allText.forEach(element => {
    const computedSize = parseFloat(getComputedStyle(element).fontSize);
    if (computedSize < 5) {
      element.style.fontSize = '5pt';
    }
  });
}
